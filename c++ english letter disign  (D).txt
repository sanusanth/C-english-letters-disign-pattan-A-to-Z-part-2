#include <iostream>
using namespace std;
int main() {
int n=10;
int x=2;
for (int i=1; i<=n; i++)
{
for (int j=1; j<=n; j++)
{
if ( j==1 || j==x)
{
cout<<"*";
}
else 
{
cout<<" ";
}
}
cout<<"\n";
if (i<=n/2)
{
x++;
}
else
{
x--;
}
}
return 0;
}
