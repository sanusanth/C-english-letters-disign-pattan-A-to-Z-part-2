#include <iostream>
using namespace std;
int main() {
int n=10;
int x=1;
for (int i=1; i<=n; i++)
{
for (int j=1; j<2*n; j++)
{
if (j==x || j==2*n-x)
{
cout<<"*";
}
else
{
cout<<" ";
}
}
cout<<"\n";
x++;
}
return 0;
}
