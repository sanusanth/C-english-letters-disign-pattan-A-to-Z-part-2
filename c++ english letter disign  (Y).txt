#include <iostream>
using namespace std;
int main() {
int n=10;
for (int i=1; i<=n; i++)
{
for (int j=1; j<=n; j++)
{
if ((j==i || j==n-i+1) && i<=n/2+1 || (j==n/2+1 && i>n/2))
{
cout<<"*";
}
else
{
cout<<" ";
}
}
// Creater By Sanusanth
cout<<"\n";
}
return 0;
}
